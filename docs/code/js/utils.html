const debounce = function (fn, delay, immediate) {
  let timer
  return function (...args) {
    if (timer) clearTimeout(timer)
    if (immediate) {
      const action = !timer
      timer = setTimeout(() => {
        fn.apply(this, args)
        timer = null
      }, delay)
      action && fn.apply(this, args)
    } else {
      timer = setTimeout(() => {
        fn.apply(this, args)
      }, delay)
    }
  }
}
